-- Вказуємо твій ID текстури (додаємо префікс rbxassetid:// для коректної роботи)
local newTextureId = "rbxassetid://134463712257662"

-- Функція для зміни текстур
local function changeAllTextures(parent)
    -- Перебираємо всі об'єкти всередині parent (Workspace або інше)
    for _, object in pairs(parent:GetDescendants()) do
        
        -- Перевіряємо тип об'єкта
        if object:IsA("Texture") or object:IsA("Decal") then
            object.Texture = newTextureId
            
        elseif object:IsA("MeshPart") then
            object.TextureID = newTextureId
            
        elseif object:IsA("SpecialMesh") then
            object.TextureId = newTextureId
            
        elseif object:IsA("BasePart") and object:FindFirstChildOfClass("Texture") == nil then
            -- Якщо хочеш змінити колір або матеріал самих блоків, можна додати тут logic
            -- Але зазвичай міняють саме об'єкти Texture/Decal
        end
    end
end

-- Запускаємо для всього світу (Workspace)
