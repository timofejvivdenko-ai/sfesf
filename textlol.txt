-- Новий Image ID, який ви хочете встановити
local newImageId = "rbxassetid://2352"

-- Функція для перебору всіх об'єктів у грі
for _, object in ipairs(game:GetDescendants()) do
    -- Перевіряємо, чи має об'єкт властивість "Image"
    -- Це стосується ImageLabel, ImageButton, Decal, Texture тощо.
    local success, hasProperty = pcall(function()
        return object.Image
    end)
    
    if success then
        object.Image = newImageId
    end
end

print("Усі ID зображень успішно змінено на: " .. newImageId)
